DROP TABLE treatments;
DROP TABLE medication_reviews;
DROP TABLE medications;

select * from medications;

CREATE TABLE medications (
  medication_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  name VARCHAR2(100),
  disease VARCHAR2(100)
);

CREATE TABLE medication_reviews (
  review_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  medication_id NUMBER REFERENCES medications(medication_id),
  patient_id NUMBER,
  review VARCHAR2(500),
  rating NUMBER(3,1)
);

CREATE TABLE treatments (
  treatment_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  diagnosis_id NUMBER,
  medication_id NUMBER REFERENCES medications(medication_id)
);
